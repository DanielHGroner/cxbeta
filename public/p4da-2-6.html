# Calculate BMI, given height and weight

# Prompt for inputs
height = float(input('Enter height (inches): '))
weight = float(input('Enter weight (pounds): '))

# calculate the BMI result
bmi = weight / (height ** 2) * 703

# display the result
print('BMI = ', bmi)
