# Translate Air Quality Index (AQI) from number to color and concern
# https://www.airnow.gov/aqi/aqi-basics/

# parallel lists of upper AQI levels, corresponding colors and concerns
levels = [50, 100, 150, 200, 300, float('inf')]
colors = ['Green', 'Yellow', 'Orange', 'Red', 'Purple', 'Maroon']
concerns = ['Good', 'Moderate', 'Unhealthy for Sensitive Groups',
            'Unhealthy', 'Very Unhealthy', 'Hazardous']
# input
level = int(input('Enter an AQI level (integer): '))

# processing
found = False
i = 0 # index of lists
while not found:
    if level <= levels[i]:
        color = colors[i]
        concern = concerns[i]
        found = True
    i += 1

# output
print('AQI color:', color)
print('AQI concern level:', concern)
